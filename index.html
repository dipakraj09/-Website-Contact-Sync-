⚙️ कोड.gs — Google Apps Script 
function syncContacts() {
  var GITHUB_TOKEN = "APNA_TOKEN_YAHAN"; // ← यहाँ अपना ghp_ token paste करो
  var GITHUB_USERNAME = "dipakraj09";        // ← अपना GitHub username
  var GITHUB_REPO = "My-Contacts";           // ← अपना repo name
  
  var url = "https://people.googleapis.com/v1/people/me/connections?personFields=names,phoneNumbers&pageSize=1000";
  
  var response = UrlFetchApp.fetch(url, {
    headers: {
      Authorization: "Bearer " + ScriptApp.getOAuthToken()
    }
  });
  
  var data = JSON.parse(response.getContentText());
  var contacts = [];
  
  if (data.connections) {
    data.connections.forEach(function(person) {
      var name = person.names ? person.names[0].displayName : "No Name";
      var phone = person.phoneNumbers ? person.phoneNumbers[0].value : "No Number";
      contacts.push({ name: name, phone: phone });
    });
  }
  
  var jsonData = JSON.stringify(contacts, null, 2);
  var ghUrl = "https://api.github.com/repos/" + GITHUB_USERNAME + "/" + GITHUB_REPO + "/contents/contacts.json";
  
  var getRes = UrlFetchApp.fetch(ghUrl, {
    headers: { Authorization: "token " + GITHUB_TOKEN },
    muteHttpExceptions: true
  });
  
  var sha = "";
  if (getRes.getResponseCode() == 200) {
    sha = JSON.parse(getRes.getContentText()).sha;
  }
  
  UrlFetchApp.fetch(ghUrl, {
    method: "PUT",
    headers: {
      Authorization: "token " + GITHUB_TOKEN,
      "Content-Type": "application/json"
    },
    payload: JSON.stringify({
      message: "Sync contacts",
      content: Utilities.base64Encode(jsonData),
      sha: sha
    })
  });
  
  Logger.log("Done! " + contacts.length + " contacts synced.");
}
